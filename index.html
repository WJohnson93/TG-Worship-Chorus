// 1. Setup the Audio Context const audioContext = new (window.AudioContext || window.webkitAudioContext)(); let audioBuffers = []; const audioFiles = [ 'https://wjohnson93.github.io/TG-Worship-Chorus/TGMelody.mp3', 'https://wjohnson93.github.io/TG-Worship-Chorus/TGH1.mp3', 'https://wjohnson93.github.io/TG-Worship-Chorus/TGH2.mp3', 'https://wjohnson93.github.io/TG-Worship-Chorus/TGPiano.mp3' ]; // 2. Load and decode audio files async function loadAudio() { for (const url of audioFiles) { const response = await fetch(url); const arrayBuffer = await response.arrayBuffer(); const audioBuffer = await audioContext.decodeAudioData(arrayBuffer); audioBuffers.push(audioBuffer); } } loadAudio(); // 3. Playback with perfect sync let sources = []; function startAudio() { // Ensure context is running (required by browsers) if (audioContext.state === 'suspended') audioContext.resume(); const startTime = audioContext.currentTime + 0.1; // Start slightly in future for precision audioBuffers.forEach((buffer) => { const source = audioContext.createBufferSource(); source.buffer = buffer; source.connect(audioContext.destination); source.start(startTime); sources.push(source); }); }
